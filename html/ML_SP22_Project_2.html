def outlier_removal(feature):
    Q1 = df[feature].quantile(0.25)
    Q2 = df[feature].quantile(0.75)
    IQR = Q2 - Q1
    upper = Q2 + (1.5*IQR)
    lower = Q1 - (1.5*IQR)
    new =  df[(df[feature]>=lower) & (df[feature]<=upper)]
    return new
